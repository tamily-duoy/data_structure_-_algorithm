# 二维数组的查找

```python

def serch_from_arr(arr,tag):

#代码块一：判断是否为空
    if arr==[]:
        return False
    
    raws=len(arr)-1
    cols=len(arr[0])-1

#代码块二：判断比较的类型是否相同
    if type(tag)==float and type(arr[0][0])==int:
        tag=int(tag)
    elif type(tag)==int and type(arr[0][0])==float:
        tag=float(tag)
    else:
        return False

#代码块三：开始比较
    raw=0
    col=cols
    while col>=0 and raw<=raws:
        if arr[raw][col]>tag:
            col-=1
        elif arr[raw][col]<tag:
            raw+=1
        else:
            return True
 
 #测试代码：
if __name__=="__main__":
    A_arr=[[1,2,"3.0"],["3.0",4,5]]
    A_tag=3
    print(serch_from_arr(A_arr,A_tag))
```
